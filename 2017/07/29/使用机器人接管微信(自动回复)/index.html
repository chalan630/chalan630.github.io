图灵机器人接口:http://www.tuling123.com/openapi/api

编写程序

这一步自己来

代码解释

引入必要的模块

import itchat
from itchat.content import *
import urllib
import json

将对方的输入进行转码:

info = msg['Text'].encode('UTF-8')

构建post所需的信息

url = 'http://www.tuling123.com/openapi/api'
data = {"key":"APIkey","info":info,"userid":"chalan630"}

发出post包

data = urllib.parse.urlencode(data).encode(encoding='UTF8')     #将data进行转码
request = urllib.request.Request(url, data)                     #发送request
response = urllib.request.urlopen(request)                      #获得response
re_info_str = response.read()                                   #提取response信息
re_info_dict = json.loads(re_info_str)                          #str转成dict

发送信息

itchat.send(re_info_dict['text'], msg['FromUserName']) #itchat信息发送
